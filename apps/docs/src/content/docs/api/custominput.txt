type Props = {
  onChange: (value: string) => void;
  value: string;
  error?: {
    message: string;
  };
};

const CustomInput = ({ value, onChange, error }: Props) => {
  return (
    <div>
      <input value={value} onChange={(e) => onChange(e.target.value)} />
      {error && <span>{error.message}</span>}
    </div>
  );
};

export default CustomInput;
